\[\forall \eps >0\ \exists N\in \N\ \forall n>p\geq N:\quad  ||f_n-f_p||_{l^2(\N)}<\eps. \label{eqn:cauchy-suite}\tag{\textasteriskcentered}\]

\end{theorem}
